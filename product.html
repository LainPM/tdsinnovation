// Smooth scroll for navigation const navLinks = document.querySelectorAll('.nav-links a'); navLinks.forEach(link => { link.addEventListener('click', function(e) { const targetId = this.getAttribute('href'); if (targetId.startsWith('#')) { e.preventDefault(); document.querySelector(targetId).scrollIntoView({ behavior: 'smooth' }); } }); }); // Contact form validation const contactForm = document.getElementById('contactForm'); contactForm.addEventListener('submit', function(e) { const name = document.getElementById('name').value.trim(); const email = document.getElementById('email').value.trim(); const message = document.getElementById('message').value.trim(); let valid = true; if (name.length < 2) { alert('Please enter your name.'); valid = false; } else if (!/^[^@\s]+@[^@\s]+\.[^@\s]+$/.test(email)) { alert('Please enter a valid email address.'); valid = false; } else if (message.length < 5) { alert('Please enter a message.'); valid = false; } if (!valid) { e.preventDefault(); } else { alert('Thank you for contacting us!'); } });
